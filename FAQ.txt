public void bg_dtg()
        {
            try
            {

                for (int i = 0; i <= dataGridFAQ.Rows.Count; i++)
                {
                    if (IsOdd(i))
                    {

                        dataGridFAQ.Rows[i].DefaultCellStyle.BackColor = Color.LightGray;
                    }
                }
            }
            catch (Exception ex)
            {
                MessageBox.Show("" + ex);
            }
        }
        public static bool IsOdd(int value)
        {
            return value % 2 != 0;
        }
 private void FAQPopUpWindow_Load(object sender, EventArgs e)
        {
            txtSearch.KeyPress += new KeyPressEventHandler(txtSearch_KeyPress);
            dataGridFAQ.CellContentClick += new DataGridViewCellEventHandler(dataGridFAQ_CellContentClick);
            dataGridFAQ.DataSource = Get_Data_From_Xml().Tables[0];
            dataGridFAQ.Columns[0].Width = 420;
            dataGridFAQ.Columns[1].Width = 450;
            bg_dtg();
            dataGridFAQ.ColumnHeadersVisible = false;
            dataGridFAQ.RowHeadersVisible = false;
            dataGridFAQ.RowTemplate.Height = 30;
           
        }

 private void txtSearch_KeyPress(object sender, KeyPressEventArgs e)
        {
            DataView dv = Get_Data_From_Xml().Tables[0].DefaultView;
            dv.RowFilter = string.Format("question like '%{0}%'", txtSearch.Text);
            dataGridFAQ.DataSource = dv.ToTable();
            lblQuestion.Text = "";
            lblAnswer.Text = "";
           // bg_dtg();
        }
private void dataGridFAQ_CellContentClick(object sender, DataGridViewCellEventArgs e)
        {
            if(e.RowIndex>=0)
            {
                DataGridViewRow row = this.dataGridFAQ.Rows[e.RowIndex];
                string Answer = row.Cells["answer"].Value.ToString();
                string Question = row.Cells["question"].Value.ToString();
                if ((Answer != null || Answer != "") && (Question != null || Question != ""))
                {
                    lblQuestion.Text = Question;
                    lblAnswer.Text = Answer;
                    Message = Answer;
                   // lblHiddenAnswer.Text = Answer.ToString();

                    
                }

            }
        }
